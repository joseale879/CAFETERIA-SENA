CREATE TABLE Usuario (
    id_usuario INT IDENTITY PRIMARY KEY,
    nombre VARCHAR(100) NOT NULL,
    identificacion VARCHAR(50) NOT NULL,
    contrasena VARCHAR(255) NOT NULL,
    rol VARCHAR(20) NOT NULL 
);
CREATE TABLE Producto (
    id_producto INT IDENTITY PRIMARY KEY,
    nombre_producto VARCHAR(100) NOT NULL,
    precio DECIMAL(10,2) NOT NULL,
);
CREATE TABLE Carrito (
    id_carrito INT IDENTITY PRIMARY KEY,
    id_usuario INT NOT NULL,
    estado VARCHAR(20) NOT NULL, 
    FOREIGN KEY (id_usuario) REFERENCES Usuario(id_usuario)
);
CREATE TABLE Detalle_Carrito (
    id_detalle_carrito INT IDENTITY PRIMARY KEY,
    id_carrito INT NOT NULL,
    id_producto INT NOT NULL,
    cantidad INT NOT NULL,
    precio_unitario DECIMAL(10,2) NOT NULL,
    FOREIGN KEY (id_carrito) REFERENCES Carrito(id_carrito),
    FOREIGN KEY (id_producto) REFERENCES Producto(id_producto)
);
CREATE TABLE Pedido (
    id_pedido INT IDENTITY PRIMARY KEY,
    id_usuario INT NOT NULL,
    estado_pedido VARCHAR(20) NOT NULL, 
    FOREIGN KEY (id_usuario) REFERENCES Usuario(id_usuario)
);
CREATE TABLE Detalle_Pedido (
    id_detalle_pedido INT IDENTITY PRIMARY KEY,
    id_pedido INT NOT NULL,
    id_producto INT NOT NULL,
    cantidad INT NOT NULL,
    precio_unitario DECIMAL(10,2) NOT NULL,
    FOREIGN KEY (id_pedido) REFERENCES Pedido(id_pedido),
    FOREIGN KEY (id_producto) REFERENCES Producto(id_producto)
);
CREATE TABLE Tiquete (
    id_tiquete INT IDENTITY PRIMARY KEY,
    id_pedido INT NOT NULL,
    codigo_tiquete VARCHAR(50) NOT NULL,
    FOREIGN KEY (id_pedido) REFERENCES Pedido(id_pedido)
);
